# 参数传递

const appRoutes: Routes = [
  //id 是一个路由参数的令牌(Token)
  { path: 'hero/:id',      component: HeroDetailComponent },
  //data 属性用来存放于每个具体路由有关的任意信息,使用resolve 守卫来获取动态数据
  {
    path: 'heroes',
    component: HeroListComponent,
    data: { title: 'Heroes List' }
  },
];

# 路由链接的激活状态

routerLinkActive的激活状态会向下级联到路由树中的每个层级，所以父子路由可能会同时激活，可以把[routerLinkActiveOptions] 绑定为 { exact: true } 表达式，这样 RouterLink 只有当 URL 与当前 URL 精确匹配时才会激活。
