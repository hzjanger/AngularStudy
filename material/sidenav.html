# 滑动侧边栏的使用

# 引入模块

import {MatSidenavModule} from '@angular/material/sidenav';

# 实现功能

  1. 点击菜单按钮，可以切换侧边栏的状态，处于显示变为隐藏，处于隐藏变为显示

  2. 当浏览器窗口缩小到一定值的时候自动隐藏，窗口大于这个值的时候自动显示

# 思路

  1. 在mat-drawer定义一个模板引用变量

  2. 由于菜单选项是定义在app-admin-header组件中的，所以接收从app-admin-header发出的事件，当app-admin-header发送这个事件时，调用MatDrawer的toggle方法，这个方法的作用是切换侧边栏的状态，处于显示变为隐藏，处于隐藏变为显示

  3. 使用BreakpointObserver的observe方法，这个方法用于得到一个可观察对象的流，只要其中指定的任何一个媒体查询中有不同的结果，它就会发出事件。详情看https://material.angular.cn/cdk/layout/overview

  4. 使用@ViewChild将MatDrawer对象与dom元素连接起来，这需要配合模板变量使用，在调用MatDrawer的close和open方法

# 代码实现

编写页面内容

<mat-drawer-container class="responsive-container" autosize>
  <mat-drawer #drawer="" class="sidenav" mode="side">
    <ng-content select="app-admin-sidebar"></ng-content>
  </mat-drawer>
  <mat-drawer-content>
    <app-admin-header (toggle)="drawer.toggle()"></app-admin-header>
    <ng-content select="router-outlet"></ng-content>
  </mat-drawer-content>
</mat-drawer-container>

组件编写

import {AfterViewInit, Component, ViewChild, ViewChildren} from '@angular/core';
import {MatDrawer} from '@angular/material';
import {BreakpointObserver, Breakpoints} from '@angular/cdk/layout';

@Component({
  selector: 'responsive-layout',
  templateUrl: './responsive.layout.html',
  styleUrls: ['./responsive.layout.scss']
})

export class ResponsiveLayout {
  showFiller = false;

  // @ts-ignore
  @ViewChild("drawer") drawer: MatDrawer;

  constructor(private breakpointObserver: BreakpointObserver) {
    //监听窗口大小变化
    breakpointObserver.observe([Breakpoints.HandsetLandscape])
      .subscribe(result => {
        if (result.matches) {
          this.drawer.close();
        } else {
          this.drawer.open();
        }
      })
  }

}
