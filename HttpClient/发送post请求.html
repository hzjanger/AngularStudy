# 使用@RequestParam接受数据

在Angular中需要设置请求头中的Content-type为application/x-www-form-urlencoded

const httpOptions = {
  headers: new HttpHeaders({
    'Content-Type':  'application/x-www-form-urlencoded',
  })
};

@Injectable({
  providedIn: 'root'
})

export class BucketService {


  url: string = '/yzm';
  constructor(private http: HttpClient) { }


  /**
   * 创建一个bucket
   * @param bucketName  bucket名称
   * @param openId  用户唯一标识
   */
  createBucket(bucketName: string, openId: string) {
    return this.http.post(`${this.url}/bucket`, `bucketName=${bucketName}&openId=${openId}`, httpOptions);
  }

}

# 使用@RequestBody接受数据

不需要设置请求头，因为在angular中请求头的Content-type默认为application/json，所以无需设置

@Injectable({
  providedIn: 'root'
})
export class FileService {

  url: string = "/yzm/hos/v1";

  /**
   * 新建文件夹
   * @param bucket 文件夹名称
   * @param path 存储路径
   */
  newFolder(bucket: string, path: string) {
    return this.http.post(`${this.url}/mkdir`, {bucket: bucket, key: path});
  }
}
